\chapter{Aplicação Fatorial}
\label{anexo:fatorial}

\begin{scriptsize}
\begin{verbatim}
	#include <stdio.h>
	
	int fat (int n) {
		if (n==1)
			return 1;
		else
			return n*fat(n-1);
	}
	
	int main() {
		int i=0;
		int result;
		for (i=0; i<10000; i++) {
			result = fat(8);
		}
		printf("Hello %d!\n", result);
		exit(0);
	}
\end{verbatim}
\end{scriptsize}
