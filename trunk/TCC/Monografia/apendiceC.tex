#include <stdio.h>

main(int argc, char *argv[])
{

  if(argc != 2) {
        printf("\n**** CHAMADA DO PROGRAMA :  ./trab1 [numero]\n");
        exit(1);
  };

  //long int i, j, N = atoi(argv[1]);
  long int i, j, N = 1000000;

  int *a = malloc(N*sizeof(int));
  if (a == NULL){
      printf("erro de alocação!!/n");
      return;
  }


  for (i = 2; i < N; i++) a[i] = 1;
  for (i = 2; i < N; i++)
    if (a[i])
      for (j = i; j<= N/i; j++) a[i*j] = 0;
  for (i = 2; i < N; i++){
    if (a[i]){
        printf("%d\n", i);
    }
  }
  exit(0);
}
